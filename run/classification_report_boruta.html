Confidence intervals for lift of recall / recall for different quantiles

ci_stats <- all_features_data %>%
  group_by(cancer_type, agg_level, quantile) %>%
  summarize(
    mean_recall = mean(recall),
    mean_lift_recall = mean(lift_recall),
    mean_precision = mean(precision),
    mean_lift_precision = mean(lift_precision),
    sd_recall = sd(recall),
    sd_lift_recall = sd(lift_recall),
    sd_precision = sd(precision),
    sd_lift_precision = sd(lift_precision),
    n_ex = n()
  ) %>%
  mutate(
    ci_lower_recall = mean_recall - 1.96 * sd_recall / n_ex,
    ci_upper_recall = mean_recall + 1.96 * sd_recall / n_ex,
    
    ci_lower_lift_recall = mean_lift_recall - 1.96 * sd_lift_recall / n_ex,
    ci_upper_lift_recall = mean_lift_recall + 1.96 * sd_lift_recall / n_ex,    
    
    ci_lower_precision = mean_precision - 1.96 * sd_precision / n_ex,
    ci_upper_precision = mean_precision + 1.96 * sd_precision / n_ex,       
    
    ci_lower_lift_precision = mean_lift_precision - 1.96 * sd_lift_precision / n_ex,
    ci_upper_lift_precision = mean_lift_precision + 1.96 * sd_lift_precision / n_ex
  )

# roc auc
roc_data <- roc_auc_data %>%
  mutate(
    train_test_diff = tr_auc - te_auc
  ) %>%

  summarize(

    sd_test_auc = sd(te_auc),
    median_test_auc = median(te_auc),
    mean_test_auc = mean(te_auc),
    train_test_diff = mean(train_test_diff)
  )

  select(cancer_type, agg_level, quantile, mean_recall, mean_lift_recall, mean_precision, mean_lift_precision)
